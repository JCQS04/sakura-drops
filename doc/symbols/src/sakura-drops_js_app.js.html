 20 /** @exports app as hlf.sakuraDrops */
 21 Namespace(pkg + 'sakuraDrops');
 22 Namespace.use(pkg + '*', function () {
 23 var app;
 24     app = sakuraDrops;
 25 // ----------------------------------------
 26 // APP GLOBALS
 27 // ----------------------------------------
 28 var canvas, context,
 29     $stopper, $exporter, $toolbar;
 30 /**
 31  * jQuery object resulting from the toolbar plugin. Has buttons including:
 32  *      #stop-animation and #export-canvas. Own id is #the-canvas-toolbar.
 33  * @requires jQuery.fn.toolbar
 34  * @type {jQuery}
 35  */
 36 app.$toolbar = $toolbar;
 37 // ----------------------------------------
 38 // APP PROCEDURES
 39 // ----------------------------------------
 40 /**
 41  * App procedure #1
 42  * @requires NamespaceJS. {@link Namespace}
 43  * @requires UnderscoreJS. {@link _}
 44  * @requires BackboneJS. {@link Backbone}
 45  * @requires Utility library. {@link hlf.util }
 46  * @requires Module library. {@link hlf.module }
 47  * @requires jQuery library. {@link hlf.jquery }
 48  */
 49 app.sketch1 = function () {
 50     var m = this.m = app.DropManager.create({
 51         num: 10,
 52         unitTest: false
 53     });
 54     $toolbar.hideButton($stopper);
 55     m.update();
 56 };
 57 /** 
 58  * On load callback for the page.
 59  */
 60 jQuery(document).ready(function ($) {
 61     // set globals
 62     canvas = app.canvas = module.Canvas.create('the-canvas');
 63     context = app.context = canvas.context;
 64     $stopper = $('#stop-animation').click(function (evt) {
 65         canvas.togglePauseAndPlay();
 66         $stopper.text(($stopper.text === 'stop') ? 'play' : 'stop');
 67         evt.preventDefault();
 68     });
 69     $exporter = $('#export-canvas').click(function (evt) {
 70         canvas.exportAsImage();
 71     });
 72     $toolbar = $('#the-canvas-toolbar').toolbar();
 73     // run
 74     app.sketch1();
 75 }); // ready
 76 }); // namespace
