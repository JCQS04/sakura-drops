 17 util.Vector = util.BaseClass({
 18     /** @lends util.Vector# */
 19     x: undefined,
 20     y: undefined,
 21     mag: undefined,
 22     _construct: function (x, y) {
 23         this.x = x || 0;
 24         this.y = y || 0;
 25         this.getMagnitude();
 26     },
 27     /**
 28      * Gets the distance between x and y.
 29      * @param {?boolean=} fresh Update protected property.
 30      * @return {number}
 31      */
 32     getMagnitude: function (fresh) {
 33         if (!this.mag || fresh) {
 34             this.mag = util.dist(this.x, this.y);
 35         }
 36         return this.mag;
 37     },
 38     /**
 39      * Adds each component of target vector, or a scalar to each component.
 40      * @param {!hlf.util.Vector|number} v Target vector or scalar.
 41      * @param {number} dir Direction: 1 or -1.
 42      */
 43     add: function (v, dir) {
 44         dir = (dir !== undefined) ? dir : 1;
 45         var isVector = v.hasOwnProperty('x');
 46         this.x += dir * (isVector ? v.x : v);
 47         this.y += dir * (isVector ? v.y : v);
 48     },
 49     /**
 50      * Multiplies each component of target vector, or a scalar to each component.
 51      * @param {!hlf.util.Vector|number} v Target vector or scalar.
 52      * @param {?number=} dir Direction: 1 or -1.
 53      */
 54     mult: function (v, dir) {
 55         v = v || 1;
 56         dir = (dir !== undefined) ? dir : 1;
 57         var isVector = v.hasOwnProperty('x');
 58         this.x *= dir * (isVector ? v.x : v);
 59         this.y *= dir * (isVector ? v.y : v);
 60     },
 61     /**
 62      * Multiply itself with the inverse of the magnitude.
 63      */
 64     normalize: function () {
 65         this.mult(1 / this.getMagnitude(true));
 66     },
 67     /** @ignore */
 68     toString: function () {
