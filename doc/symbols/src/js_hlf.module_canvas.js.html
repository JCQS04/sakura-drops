1 /**
  2  * @fileoverview Canvas module.
  3  * @author peng@pengxwang.com (Peng Wang)
  4  */
  5 /** @exports util as hlf.util */
  6 /** @exports module as hlf.module */
  7 _.namespace(pkg + 'module');
  8 _.using(pkg + '*', function () {
  9 /**
 10  * @class A custom wrapper class for the HTMLCanvasElement and its API. Its
 11  *      purpose is to simplify common tasks and provide an object-oriented API
 12  *      for the otherwise declarative canvas API, and to add a more robust 
 13  *      handling of canvas state and canvas-wide events.
 14  * @augments hlf.util.BaseClass
 15  * @augments hlf.module.EventMixin
 16  * @property {Element} canvas Associated canvas element.
 17  * @property {CanvasRenderingContext} context Associated canvas context/API.
 18  * @property {Object} _plotter Metaphorical pen.
 19  * @property {Array Object} _plotters Collection of pens, useful for changing contexts.
 20  * @property {hlf.module.Canvas.AnimationState} animationState
 21  * @property {Array number} _animationTimers List of interval ids.
 22  * @property {Array Object} _animations List of generally a step lambda and settings.
 23  * @property {boolean} anticlockwise Setting for all future arcs drawn.
 24  * @param {!string} id Canvas DOM element id.
 25  * @param {?string=} type Canvas context/API type.
 26  */
 27 module.Canvas = util.BaseClass(util.extend({
 28   /** @lends module.Canvas# */
 29   // ----------------------------------------
 30   // PROPERTIES
 31   // ----------------------------------------
 32   canvas: undefined,
 33   context: undefined,
 34   _plotter: undefined,
 35   _plotters: undefined,
 36   animationState: undefined,
 37   _animationTimers: undefined,
 38   _animations: undefined,
 39   anticlockwise: undefined,
 40   // ----------------------------------------
 41   // METHODS
 42   // ----------------------------------------
 43   /**
 44    * Gets the element by id and sets the dimension properties via existing 
 45    *      CSS properties.
 46    */
 47   _construct: function (id, type) {
 48     this.canvas = document.getElementById(id);
 49     if (!this.canvas.getAttribute('width')) {
 50       this.canvas.setAttribute('width', this.canvas.clientWidth);
 51       this.canvas.setAttribute('height', this.canvas.clientHeight);
 52     }
 53     this.context = this.canvas.getContext(type) || this.canvas.getContext('2d');
 54     this._plotter = { 'pos': { 'x': 0, 'y': 0 } };
 55     this._plotters = [];
 56     this._animations = [];
 57     this._animationTimers = [];
 58     this.animationState = module.Canvas.AnimationState.STOPPED;
 59     this.anticlockwise = true;
 60   },
 61   //---------------------------------------
 62   // ACCESSORS
 63   //---------------------------------------
 64   /** Accessor. */
 65   getWidth: function () {
 66     return this.canvas.width;
 67   },
 68   /** Accessor. */
 69   getHeight: function () {
 70     return this.canvas.height;
 71   },
 72   /** Plotter position accessor. */
 73   getX: function () {
 74     return this._plotter.pos.x;
 75   },
 76   /** Plotter position accessor. */
 77   getY: function () {
 78     return this._plotter.pos.y;
 79   },
 80   /** For arcs, converts counterclockwise to -1. */
 81   getAngDir: function () {
 82     return this.anticlockwise ? -1 : 1;
 83   },
 84   //---------------------------------------
 85   // PLOTTING
 86   //---------------------------------------
 87   /** 
 88    * Update pen position.
 89    * @param {!number} x Horizontal offset.
 90    * @param {!number} y Vertical offset.
 91    * @param {?boolean=} isShift Increment from current offset.
 92    * @return {hlf.module.Canvas} Chains.
 93    */
 94   movePlotter: function (x, y, isShift) {
 95     isShift = isShift || false;
 96     this._plotter.pos = isShift ? 
 97       {'x': this._plotter.pos.x + x, 'y': this._plotter.pos.y + y} :
 98       {'x': x, 'y': y};
 99     if (!isShift) {
100       this.context.moveTo(x, y);
101     }
102     return this;
103   },
104   shiftPlotter: function (x, y) {
105     return this.movePlotter(x, y, true);
106   },
107   /** 
108    * Shorthand for drawing a sequence of something.
109    * @param {?Object=} opt Parameters.
110    * @param {function(Object, CanvasRenderingContext, number)} 
111    *      cb Step callback.
112    * @return {hlf.module.Canvas} Chains.
113    */
114   sequence: function (opt, cb) {
115     var plotter = this._plotter,
116       context = this.context;
117     opt = opt || {'x': 10, 'y': 10, 'num': 10};
118     for (var idx = 0; idx < opt.num; idx += 1) {
119       cb({
120         'x': plotter.pos.x + opt.x * idx,
121         'y': plotter.pos.y + opt.y * idx
122       }, context, idx);
123     }
124     return this;
125   },
126   /** 
127    * Shortcut to drawing a circle, and by default sets fill and stroke.
128    * @param {!number} rad Radius.
129    * @param {?boolean=} noFill Defaults to false.
130    * @param {?boolean=} noStroke Defaults to false.
131    */
132   circle: function (rad, noFill, noStroke) {
133     var plotter = this._plotter,
134       context = this.context;
135     context.beginPath();
136     context.arc(plotter.pos.x + rad + 1, 
137           plotter.pos.y + rad + 1, 
138           rad, 0, Math.PI * 2, 
139           this.anticlockwise);
140     if (!noFill) { context.fill(); }
141     if (!noStroke) { context.stroke(); }
142     context.closePath();
143   },
144   /**
145    * Shortcut to drawing a straight line.
146    * @param {!number} x1
147    * @param {!number} y1
148    * @param {!number} x2
149    * @param {!number} y2
150    */
151   line: function (x1, y1, x2, y2) {
152     var context = this.context;
153     context.beginPath();
154     context.lineTo(x1, y1);
155     context.lineTo(x2, y2);
156     context.stroke();
157     context.closePath();
158   },
159   /**
160    * Shortcut to drawing a straight line between nodes.
161    * @param {!Object number} pos1
162    * @param {!Object number} pos2
163    */
164   connect: function (pos1, pos2) {
165     this.line(pos1.x, pos1.y, pos2.x, pos2.y);
166   },
167   /**
168    * Shortcut to drawing an arc.
169    * @param {!number} x
170    * @param {!number} y
171    * @param {!number} r
172    * @param {!number} start
173    * @param {!number} end
174    * @param {!number} offset Hack for rotation.
175    */
176   arc: function (x, y, r, start, end, offset) {
177     offset = offset || 0;
178     this.context.arc(x, y, r, start + offset, end + offset, this.anticlockwise);
179   },
180   /**
181    * Save overall context and state, wraps native API.
182    */
183   save: function () {
184     this._plotters.push(this._plotter);
185     this.context.save();
186   },
187   /**
188    * Restore overall context and state, wraps native API.
189    */
190   restore: function () {
191     this._plotter = this._plotters.pop();
192     this.context.restore();
193   },
194   //---------------------------------------
195   // ANIMATION
196   //---------------------------------------
197   /** 
198    * Sets up an animation. Animation API.
199    * @param {?Object=} opt Parameters.
200    * @param {function} cb Frame callback.
201    * @param {?int=} duration Default is infinite.
202    * @param {?int=} idx Id of animation and timer. Always should be same.
203    * @return {number} The id.
204    */
205   animate: function (opt, cb, duration, idx) {
206     opt = opt || {'fps': 24};
207     var anim = {'opt': opt, 'cb': cb};
208     if (duration) {
209       anim.duration = duration;
210     }
211     this._startAnimation(anim, idx);
212     if (!idx) {
213       this._animations.push(anim);
214       return this._animations.length - 1; 
215     }
216   },
217   /** 
218    * Resets a timer, updates the index slot, and sets state to playing.
219    * @param {!Object} anim Animation.
220    * @param {?number=} idx Id of animation.
221    * @param {?int=} duration Default is infinite.
222    * @requires hlf.util.millisPerFrame
223    */
224   _startAnimation: function (anim, idx) {
225     var _this = this;
226     idx = idx || this._animationTimers.length;
227     if (this._animationTimers[idx]) { // reset
228       clearInterval(this._animationTimers[idx]);
229     }
230     if (anim.duration) {
231       var start = this.millis(), 
232         elapsed = 0,
233         complete = false;
234       this._animationTimers[idx] = setInterval(function () {
235         elapsed = _this.millis() - start;
236         if (elapsed >= anim.duration) {
237           complete = true;
238         } else {
239           _this.clear();
240           // console.log('frame');
241         }
242         anim.cb(elapsed, complete);
243       }, util.millisPerFrame(anim.opt.fps));  
244     } else {
245       this._animationTimers[idx] = setInterval(function () {
246         _this.clear();
247         anim.cb();
248       }, util.millisPerFrame(anim.opt.fps));  
249     }
250     this.animationState = module.Canvas.AnimationState.PLAYING;
251   },
252   /** 
253    * Clears the timer and sets state to paused.
254    * @param {!number} idx Id of timer.
255    */
256   _stopAnimation: function (idx) {
257     clearInterval(this._animationTimers[idx]);
258     this.animationState = module.Canvas.AnimationState.PAUSED;
259   },
260   /** 
261    * Removes the timer and animation from the respective indexes.
262    * @param {!number} idx Id of animation and timer.
263    * @todo Return to original frame.
264    */
265   _deleteAnimation: function (idx) {
266     this._stopAnimation(idx);
267     this._animationTimers.splice(idx, 1);
268     this._animations.splice(idx, 1);
269     this.animationState = module.Canvas.AnimationState.STOPPED;
270   },
271   /**
272    * Temporary, for now.
273    */
274   clearAnimations: function () {
275     for (var i = 0, l = this._animationTimers.length; i < l; i += 1) {
276       clearInterval(this._animationTimers[i]);
277     }
278     this._animations = [];
279     this._animationTimers = [];
280   },
281   /** 
282    * State controller. Animation API.
283    * @param {!hlf.module.Canvas.AnimationState} key 
284    * @param {?number=} idx Id of animation and timer.
285    */
286   changeAnimationStateTo: function (key, idx) {
287     if (!this.AnimationState[key]) {
288       return;
289     }
290     var anim = this._animations[idx || this._animations.length - 1];
291     switch (this.AnimationState[key]) {
292       case module.Canvas.AnimationState.PLAYING:
293         this._startAnimation(anim, idx);
294         break;
295       case module.Canvas.AnimationState.PAUSED:
296         this._stopAnimation(idx);
297         break;
298       case module.Canvas.AnimationState.STOPPED:
299         this._deleteAnimation(idx);
300         break;
301     }
302   },
303   /**
304    * Play command. Animation API.
305    * @param {?number=} idx Id of animation and timer.
306    */
307   playAnimation: function (idx) {
308     idx = idx || this._animations.length - 1;
309     this.changeAnimationStateTo('PLAYING', idx);
310   },
311   /**
312    * Pause command. Animation API.
313    * @param {?number=} idx Id of animation and timer.
314    */
315   pauseAnimation: function (idx) {
316     idx = idx || this._animations.length - 1;
317     this.changeAnimationStateTo('PAUSED', idx);
318   },
319   /**
320    * Smart play command. Animation API.
321    * @param {?number=} idx Id of animation and timer.
322    */
323   togglePauseAndPlay: function (idx) {
324     if (this.animationState === module.Canvas.AnimationState.PLAYING) {
325       this.changeAnimationStateTo('PAUSED', idx);
326     } else {
327       this.changeAnimationStateTo('PLAYING', idx);
328     }
329   },
330   /**
331    * Draws a fade over the screen. For blanking use {@link #clear}.
332    * @param {string} color 
333    */
334   background: function (color) {
335     this.context.save();
336     this.context.fillStyle = color;
337     this.context.fillRect(0, 0, this.getWidth(), this.getHeight());
338     this.context.restore();
339   },
340   /**
341    * Simple clearing rectangle over everything.
342    */
343   clear: function () {
344     this.context.clearRect(0, 0, this.getWidth(), this.getHeight());
345   },
346   /**
347    * Simple way to get current time signature.
348    * @return {int}
349    */
350   millis: function () {
351     return new Date().getTime();
352   },
353   // ----------------------------------------
354   // MISC
355   // ----------------------------------------
356   /**
357    * Saved as png by default, and is lossless. Opens a new window / tab
358    *      with the url being the resulting 'dataURL' of the image.
359    */
360   exportAsImage: function () {
361     window.open(this.canvas.toDataURL('image/png'));
362   },
363   /** @ignore */
364   toString: function () {
365     return pkg + 'module.Canvas';
366   }
367 }, module.EventMixin));
368 /**
369  * Enum: PLAYING, PAUSED, STATIC.
370  * @type {Object int}
371  * @static 
372  */
373 module.Canvas.AnimationState = {
374   PLAYING: 1,
375   PAUSED: 2,
376   STOPPED: 3
377 };
378 /**
379  * @class Event handling for canvas using jQuery event API.
380  * @requires jQuery
381  * @requires For now, requires the class to have a $canvas property.
382  */
383 util.CanvasEventMixin = {
384   /** @lends util.CanvasEventMixin# */
385   /** Sets up responsiveness for mouse events. */
386   bindMouse: function () {
387     var _this = this;
388     this.$canvas.bind({
389       mousemove: function (evt) { _this.onMouseMove(evt); evt.stopPropagation(); },
390       mousedown: function (evt) { _this.onMouseDown(evt); evt.stopPropagation(); },
391       mouseup: function (evt) { _this.onMouseUp(evt); evt.stopPropagation(); },
392       mouseenter: function (evt) { _this.onMouseEnter(evt); evt.stopPropagation(); },
393       mouseleave: function (evt) { _this.onMouseLeave(evt); evt.stopPropagation(); }
394     });
395   },
396   /**#@+
397      Event handler.
398      @param {MouseEvent} evt
399   */
400   onMouseMove: function (evt) {},
401   onMouseDown: function (evt) {},
402   onMouseUp: function (evt) {},
403   onMouseEnter: function (evt) {},
404   onMouseLeave: function (evt) {}
405   /**#@-*/
406 };
407 }); // namespace
