1 _.namespace(pkg + 'util');
  2 _.using(pkg + '*', function () {
  3 //---------------------------------------
  4 // CONVERSIONS
  5 //---------------------------------------
  6 /** 
  7  * Conversion function.
  8  * @param {!number} num Frames per second.
  9  * @return {number} Millis for each frame.
 10  */
 11 util.millisPerFrame = function (num) {
 12   return parseInt(1000 / num, 10);
 13 };
 14 //---------------------------------------
 15 // CALCULATION
 16 //---------------------------------------
 17 /** 
 18  * Pythagoras theorem.
 19  * @param {!number} a
 20  * @param {!number} b
 21  * @return {number} c
 22  */
 23 util.dist = function (a, b) {
 24   return Math.sqrt((a * a) + (b * b));
 25 };
 26 /**
 27  * Limits a number to the markers.
 28  * @param {!number} num
 29  * @param {!number} min
 30  * @param {!number} max
 31  * @return {number}
 32  */
 33 util.constrain = function (num, min, max) {
 34   return Math.min(Math.max(num, min), max);
 35 };
 36 /**
 37  * Gets a random point between the two markers.
 38  * @param {!number} to
 39  * @param {number=} from
 40  * @return {number}
 41  */
 42 util.simpleRandom = function (to, from) {
 43   from = (from !== undefined) ? from : 0;
 44   return (to - from) * Math.random() + from;
 45 };
 46 /**
 47  * Returns either -1 or 1.
 48  * @return {number} 
 49  */
 50 util.simpleDirection = function () {
 51   return Math.random() < 0.5 ? -1 : 1;
 52 };
 53 /** 
 54  * A simple pseudo-random generator, where the random component takes up
 55  *      <strong>1 / buffer</strong> of the result.
 56  * @param {!number} num Base number.
 57  * @param {number=} buffer Ratio to keep outside of random, defaults to 1.
 58  * @return {number}
 59  * @example util.bufferedRandom(100) // more likely to return 50+
 60  */
 61 util.bufferedRandom = function (num, buffer) {
 62   buffer = (buffer !== undefined) ? buffer : 1;
 63   return num / (buffer + 1) * (Math.random() + buffer);
 64 };
 65 /**
 66  * Extension of {@link hlf.util.bufferedRandom} exponentiating random component.
 67  * @param {!number} num Base number.
 68  * @param {number=} buffer Ratio to keep outside of random, defaults to 1.
 69  * @param {number=} pow Power, defaults to 2.
 70  * @return {number}
 71  * @see {hlf.util.bufferedRandom}
 72  */
 73 util.curvingBufferedRandom = function (num, buffer, pow) {
 74   buffer = (buffer !== undefined) ? buffer : 1;
 75   pow = (pow !== undefined) ? pow : 2;
 76   return num / (buffer + 1) * (Math.pow(Math.random(), pow) + buffer);
 77 };
 78 // ----------------------------------------
 79 // EASING
 80 // ----------------------------------------
 81 /**
 82  * Uses the ease-in-out-cubic easing equation to get the d(change in value).
 83  * @param {!int} t Current time, in millis.
 84  * @param {!number} b Beginning value.
 85  * @param {!number} c Overall change in value.
 86  * @param {!int} d Duration, in millis.
 87  * @return {number} Size of current step.
 88  */
 89 util.easeInOutCubic = function (t, b, c, d) {
 90   if ((t /= d / 2) < 1) {
 91     return c / 2 * t * t * t + b;
 92   }
 93   return c / 2 * ((t -= 2) * t * t + 2) + b; 
 94 };
 95 }); // namespace
