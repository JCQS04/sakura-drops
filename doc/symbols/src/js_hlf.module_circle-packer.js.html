 52 /**
 53  * @class Representation of a circle.
 54  * @augments hlf.util.BaseClass
 55  * @property {number} rad Radius, default is 0.
 56  * @property {Object<number>} pos Position, default is 0, 0.
 57  * @param {?number=} rad
 58  * @param {?Object=} pos
 59  */
 60 util.Circle = util.BaseClass({
 61     /** @lends util.Circle# */
 62     rad: undefined,
 63     pos: undefined,
 64     _construct: function (rad, pos) {
 65         this.rad = rad || 0;
 66         this.pos = pos || {x: 0, y: 0};
 67     },
 68     /**
 69      * Move with a vector, in distance and direction.
 70      * @param {!hlf.util.Vector|Object} v Modifying set of components, 
 71      *      typically a vector.
 72      */
 73     shiftAlong: function (v) {
 74         this.pos.x += v.x;
 75         this.pos.y += v.y;
 76     },
 77     /**
 78      * Move with a vector, in distance and direction.
 79      * @param {!hlf.util.Vector|Object} v Modifying set of components,
 80      *      typically a vector.
 81      */
 82     shiftAgainst: function (v) {
 83         this.pos.x -= v.x;
 84         this.pos.y -= v.y;
 85     },
 86     /**
 87      * Update to position to fit within the view as defined by the view model 
 88      *      delegate accessors for view bounds.
 89      * @param {!hlf.util.ViewModel} vm View model with the required accessors.
 90      * @deprecated until I refine this.
 91      */
 92     updateToView: function (vm) {
 93         this.pos.x = Math.min(vm.getBX(this.rad), this.pos.x);
 94         this.pos.y = Math.min(vm.getBY(this.rad), this.pos.y);
 95         this.pos.x = Math.max(0, this.pos.x);
 96         this.pos.y = Math.max(0, this.pos.y);
 97     },
 98     /** @ignore */
 99     toString: function () {
100         return pkg + 'util.Circle';
101     }

103 /**
104  * @class This is a port of moocirclepack. The distance methods have been
105  *      relocated from the Circle object.
106  * @augments hlf.util.BaseClass
107  * @augments hlf.module.EventMixin
108  * @requires hlf.util.CirclePackerViewModel
109  * @requires hlf.util.Circle
110  * @requires hlf.util.dist
111  * @author Joshua Gross <a href="mailto:josh@unwieldy.net">email</a> | 
112  *      <a href="http://unwieldy.net">site</a>
113  * @property {Array hlf.util.Circle} circles 
114  * @property {util.CirclePackerViewModel} vm View model.
115  * @property {Object number} origin Attractor position.
116  * @property {number} damping Modifier for attracting distance to grow and cause
117  *      settling effect.
118  * @property {int} runner Interval id.
119  * @property {int} passes Goal.
120  * @property {int} pass Current.
121  * @param {!Object mixed|Array hlf.util.Circle} arg1 View model, or circles.
122  * @param {Object number=} arg2 If arg1 is circles, origin.
123  * @param {Object number=} arg3 If arg1 is circles, passes.
124  */
125 module.CirclePacker = util.BaseClass(util.extend({
126     /** @lends module.CirclePacker# */
127     circles: undefined,
128     vm: undefined,
129     origin: undefined,
130     damping: undefined,
131     runner: undefined,
132     passes: undefined,
133     pass: undefined,
134     /**
135      * Circle packer can work as an independent canvas agent through being 
136      *      passed a view model, or it can work under a canvas agent / node 
137      *      manager by having its circles map to the nodes.
138      * @param {Array} arguments
139      */
140     _construct: function () {
141         if (arguments.length > 1) {
142             this.circles = arguments[0];
143             this.origin = arguments[1];
144             this.passes = arguments[2];
145         } else if (arguments.length === 1) {
146             this.vm = arguments[0];
147             this.origin = this.vm.getOrigin();
148             this.circles = [];
149             this.passes = this.vm.N_PASS;
150             var num = this.vm.N_BASE,
151                 rad, i;
152             for (i = 0; i < num; i++) {
153                 rad = this.vm.getRad(i / num);
154                 pos = this.vm.getPos(i / num, rad);
155                 this.circles[i] = util.Circle.create(rad, pos);
156             }
157         }
158     },
159     /**
160      * Attach a handler to this event to draw on a circle packer pass. 
161      * @name module.CirclePacker#drawingSocket
162      * @event
163      * @param {Event} e
164      */
165     /**
166      * @name module.CirclePacker#willSettle
167      * @event
168      * @param {Event} e
169      */
170     /**
171      * @name module.CirclePacker#didSettle
172      * @event
173      * @param {Event} e
174      */
175     /**
176      * Public API. Unfortunately, in browsers, it is currently impractical to run a for-loop
177      *      on the circle packing algorithm. Instead, we do it with a timer which,
178      *      while more sloppy, allows the processor to breathe and prevents the browser
179      *      from locking up.
180      * @param {?int=} pass Starting pass.
181      */
182     run: function (pass) {
183         var _this = this;
184         this.pass = pass ? pass : 0;
185         this.damping = (pass ? Math.pow(0.98, pass) : 1) * 0.1; // TODO - refactor into const
186         if (this.runner) {
187             clearInterval(this.runner);
188         }
189         this.runner = setInterval(function () {
190             _this.trigger('drawingSocket');
191             _this._iterator(_this.pass += 1);
192             if (_this.damping < 0.007) {
193                 clearInterval(_this.runner);
194                 _this.trigger('didSettle');
195             }
196         }, 30);
197         this.trigger('willSettle');
198     },
199     /** 
200      * Main calculator and iterator.
201      * @param {?int=} pass Current.
202      * @todo Fix porting hack for Function.bind
203      */
204     _iterator: function (pass) {
205         var ci, cj, dx, dy, d, r, i, j,
206             v = util.Vector.create(),
207             _this = this;
208         this.circles.sort(function (c1, c2) {
209             return _this._compare(c1, c2);
210         });
211         for (i = 0, l = this.circles.length; i < l; i += 1) {
212             ci = this.circles[i];
213             for (j = i + 1; j < l; j += 1) {
214                 if (i == j) {
215                     continue;
216                 }
217                 cj = this.circles[j];
218                 dx = cj.pos.x - ci.pos.x;
219                 dy = cj.pos.y - ci.pos.y;
220                 d = (dx * dx) + (dy * dy);
221                 r = cj.rad + ci.rad;
222                 if (d < (r * r - 0.01)) {
223                     v.x = dx;
224                     v.y = dy;
225                     v.normalize();
226                     v.mult((r - Math.sqrt(d)) * 0.5);
227                     cj.shiftAlong(v);
228                     ci.shiftAgainst(v);
229                 }
230             }
231         }
232         this.damping *= 0.98;
233         this._contract(this.damping);
234     },
235     /**
236      * Tweaks the circle's position based on the current damping value.
237      * @param {!number} damping Current.
238      */
239     _contract: function (damping) {
240         if (damping < 0.01) {
241             return;
242         }
243         var x, y, ci;
244         for (var i = 0, l = this.circles.length; i < l; i += 1) {
245             ci = this.circles[i];
246             x = (ci.pos.x - this.origin.x) * damping;
247             y = (ci.pos.y - this.origin.y) * damping;
248             ci.shiftAgainst(util.Vector.create(x, y));
249         }
250     },
251     /** 
252      * Compare distance vector with radius vector.
253      * @param {!hlf.util.Circle} c1 Circle to pack.
254      * @param {!hlf.util.Circle} c2 Circle for reference.
255      * @return {boolean}
256      */
257     _compare: function (c1, c2) {
258         var dist1 = this._distanceTo(c1.pos, this.origin);
259         var dist2 = this._distanceTo(c2.pos, this.origin);
260         if (dist1 < dist2) {
261             return 1;
262         } else if (dist1 > dist2) {
263             return -1;
264         } else {
265             return 0;
266         }
267     },
268     /**
269      * Gets the distance from the differences between the positions' components. 
270      * @param {Object number} pos1
271      * @param {Object number} pos2
272      */
273     _distanceTo: function(pos1, pos2) {
274         var dx = pos1.x - pos2.x,
275             dy = pos1.y - pos2.y;
276         return util.dist(dx, dy);
277     },
278     /** @ignore */
279     toString: function () {
280         return pkg + 'module.CirclePacker';
281     }
282 }, module.EventMixin));
283 }); // namespace
