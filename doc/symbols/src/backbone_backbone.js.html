 37   // A module that can be mixed in to *any object* in order to provide it with
 38   // custom events. You may `bind` or `unbind` a callback function to an event;
 39   // `trigger`-ing an event fires all callbacks in succession.
 40   //
 41   //     var object = {};
 42   //     _.extend(object, Backbone.Events);
 43   //     object.bind('expand', function(){ alert('expanded'); });
 44   //     object.trigger('expand');
 45   //
 46   Backbone.Events = {
 47 
 48     // Bind an event, specified by a string name, `ev`, to a `callback` function.
 49     // Passing `"all"` will bind the callback to all events fired.
 50     bind : function(ev, callback) {
 51       var calls = this._callbacks || (this._callbacks = {});
 52       var list  = this._callbacks[ev] || (this._callbacks[ev] = []);
 53       list.push(callback);
 54       return this;
 55     },
 56 
 57     // Remove one or many callbacks. If `callback` is null, removes all
 58     // callbacks for the event. If `ev` is null, removes all bound callbacks
 59     // for all events.
 60     unbind : function(ev, callback) {
 61       var calls;
 62       if (!ev) {
 63         this._callbacks = {};
 64       } else if (calls = this._callbacks) {
 65         if (!callback) {
 66           calls[ev] = [];
 67         } else {
 68           var list = calls[ev];
 69           if (!list) return this;
 70           for (var i = 0, l = list.length; i < l; i++) {
 71             if (callback === list[i]) {
 72               list.splice(i, 1);
 73               break;
 74             }
 75           }
 76         }
 77       }
 78       return this;
 79     },
 80 
 81     // Trigger an event, firing all bound callbacks. Callbacks are passed the
 82     // same arguments as `trigger` is, apart from the event name.
 83     // Listening for `"all"` passes the true event name as the first argument.
 84     trigger : function(ev) {
 85       var list, calls, i, l;
 86       if (!(calls = this._callbacks)) return this;
 87       if (list = calls[ev]) {
 88         for (i = 0, l = list.length; i < l; i++) {
 89           list[i].apply(this, Array.prototype.slice.call(arguments, 1));
 90         }
 91       }
 92       if (list = calls['all']) {
 93         for (i = 0, l = list.length; i < l; i++) {
 94           list[i].apply(this, arguments);
 95         }
 96       }
 97       return this;
 98     }

449     // Internal implementation of adding a single model to the set, updating
450     // hash indexes for `id` and `cid` lookups.
451     _add : function(model, options) {
452       options || (options = {});
453       if (!(model instanceof Backbone.Model)) {
454         model = new this.model(model);
455       }
456       var already = this.getByCid(model);
457       if (already) throw new Error(["Can't add the same model to a set twice", already.id]);
458       this._byId[model.id] = model;
459       this._byCid[model.cid] = model;
460       model.collection = this;
461       var index = this.comparator ? this.sortedIndex(model, this.comparator) : this.length;
462       this.models.splice(index, 0, model);
463       model.bind('all', this._boundOnModelEvent);
464       this.length++;

466       return model;

472       options || (options = {});

474       if (!model) return null;
475       delete this._byId[model.id];
476       delete this._byCid[model.cid];
477       delete model.collection;
478       this.models.splice(this.indexOf(model), 1);
479       model.unbind('all', this._boundOnModelEvent);
480       this.length--;
481       if (!options.silent) this.trigger('remove', model);
482       return model;
483     },

493           }

495           break;
496         case 'error':
497           this.trigger('error', model, error);
498           break;
499       }

573     //       'click .button':     'save'
574     //     }

578     // Omitting the selector binds the event to `this.el`.
579     // `"change"` events are not delegated through the view because IE does not

581     handleEvents : function(events) {
582       $(this.el).unbind();
583       if (!(events || (events = this.events))) return this;
584       for (var key in events) {
585         var methodName = events[key];
586         var match = key.match(eventSplitter);
587         var eventName = match[1], selector = match[2];
588         var method = _.bind(this[methodName], this);
589         if (selector === '' || eventName == 'change') {
590           $(this.el).bind(eventName, method);
591         } else {
592           $(this.el).delegate(selector, eventName, method);
593         }
594       }

641   // Override this function to change the manner in which Backbone persists
642   // models to the server. You will be passed the type of request, and the
643   // model in question. By default, uses jQuery to make a RESTful Ajax request
644   // to the model's `url()`. Some possible customizations could be:
645   //
646   // * Use `setTimeout` to batch rapid-fire updates into a single request.
647   // * Send up the models as XML instead of JSON.
648   // * Persist models via WebSockets instead of Ajax.
649   //
650   // Turn on `Backbone.emulateHttp` in order to send `PUT` and `DELETE` requests
651   // as `POST`, with an `_method` parameter containing the true HTTP method.
652   // Useful when interfacing with server-side languages like **PHP** that make
653   // it difficult to read the body of `PUT` requests.
654   Backbone.sync = function(method, model, success, error) {
655     var sendModel = method === 'create' || method === 'update';
656     var data = sendModel ? {model : JSON.stringify(model)} : {};
657     var type = methodMap[method];
658     if (Backbone.emulateHttp && (type === 'PUT' || type === 'DELETE')) {
659       data._method = type;
660       type = 'POST';
661     }
662     $.ajax({
663       url       : getUrl(model),
664       type      : type,
665       data      : data,
666       dataType  : 'json',
667       success   : success,
668       error     : error
669     });
670   };
