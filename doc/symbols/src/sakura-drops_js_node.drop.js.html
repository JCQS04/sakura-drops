  3  * @author peng@pengxwang.com (Peng Wang)

  5 // ----------------------------------------
  6 // OVERHEAD
  7 // ----------------------------------------
  8 Namespace(pkg + 'sakuraDrops');
  9 Namespace.use(pkg + '*', function () {
 10 var app = sakuraDrops, 
 11     C = app.constants.DROP_NODE, // custom
 12     M = app.Math, 
 13     canvas, context,
 14     requireCanvasGlobals = function () {
 15         if (!canvas) {
 16             canvas = app.canvas;
 17             context = canvas.context;
 18         }  
 19     };
 20 // ----------------------------------------
 21 // CLASS
 22 // ----------------------------------------
 23 /**
 24  * @class The most common type of nodes. Each node can have segmented inner and 
 25  *      outer arcs, though the visual results vary based on chance and parameterization. 
 26  *      Having an shorter and smaller arcs generally don't have inner rings. 
 27  *      Outer segmentation is generally due to luck, which also affects size. 
 28  *      Segments are separated by short, uniform breaks.
 29  * @augments hlf.sakuraDrops.BaseNode
 30  * @property {Object} innerRing Collection of inner ring properties. Inner ring
 31  *      has shorter radius, thinner stroke, bigger gaps, and links to outer ring 
 32  *      at the ends. A base radius requirement is needed to have an inner ring. 
 33  *      Inner rings also have a rotation offset from the outer ring. The end 
 34  *      structure of the inner ring also decides how it bridges with the outer ring.
 35  *      Properties:
 36  *      <br/>rad Based on innerRad constant.
 37  *      <br/>lineWidth Based on innerLineWidth constant.
 38  *      <br/>angStart Based on Math.QTR_PI.
 39  *      <br/>andEnd Based on Math.QTR_PI.
 40  *      <br/>bridgeBend Based on innerBendMin and innerBendMax constants.
 41  *      <br/>hasStartBridge Always true, for now.
 42  *      <br/>hasEndBridge True only if segments run the full ring length.
 43  * @property {Array Object} outerSegments Segment data for outer ring. Properties:
 44  *      <br/>angStart
 45  *      <br/>angEnd
 46  * @property {Array Object} innerSegments Segment data for inner ring. Properties:
 47  *      <br/>angStart
 48  *      <br/>angEnd
 49  */
 50 app.DropNode = app.BaseNode.extend({
 51     /** @lends app.DropNode# */
 52     innerRing: undefined,
 53     outerSegments: undefined,
 54     innerSegments: undefined,
 55     // ----------------------------------------
 56     // ACCESSORS
 57     // ----------------------------------------
 58     /**#@+
 59        Accessor.
 60        @return {boolean}
 61        @see radWithInnerRing, outerSegmentsChance, innerSegmentsChance constants.
 62      */
 63     hasInnerRing: function () {
 64         return this.innerRing || (this.getArcLen() > M.PI && 
 65             this.rad > C.radWithInnerRing);
 66     },
 67     hasOuterSegments: function () {
 68         return this.outerSegments || this.luck <= C.outerSegmentsChance;
 69     },
 70     hasInnerSegments: function () {
 71         return this.innerRing && (this.innerSegments || 
 72             this.luck <= C.innerSegmentsChance);
 73     },
 74     /**#@-*/
 75     // ----------------------------------------
 76     // SETUP
 77     // ----------------------------------------
 78     /**
 79      * Setup for the custom node traits. Parameters are built through chance and 
 80      *      parameterization. 
 81      * @see #_generateSegments
 82      * @see radWithInnerRing, innerBuffer, innerRad, innerBendMin, innerBendMax 
 83      *      constants
 84      * @see hlf.util.simpleRandom
 85      */
 86     didCreate: function () {
 87         // deviation from base rad
 88         requireCanvasGlobals();
 89         var d = M.sqrt(C.radWithInnerRing / this.rad),
 90             r = util.simpleRandom(-1, 1) * C.innerBuffer,
 91             dir = canvas.getAngDir(),
 92             _this = this;
 93         if (this.hasInnerRing()) {
 94             this.innerRing = {
 95                 rad: _this.rad * C.innerRad * d,
 96                 lineWidth: M.sqrt(_this.lineWidth) * C.innerLineWidth * d,
 97                 angStart: _this.angStart - d + r * M.QTR_PI * dir, 
 98                 angEnd: _this.angEnd + d + r * M.QTR_PI * dir,
 99                 bridgeBend: d * util.simpleRandom(C.innerBendMin, C.innerBendMax),
100                 hasStartBridge: true,
101                 hasEndBridge: true
102             };
103         }
104         this._generateSegments();
105     },
106     /**
107      * Super method for generating arc segment data for both outer and inner
108      *      rings. Uses inner functions as sub procedures to abstract
109      *      common tasks for producing both types of segments. 
110      * @see innerSegmentsMax, outerSegmentsMax, innerSegmentBreak, outerSegmentBreak 
111      *      segmentFluxMin, segmentFluxMax constants
112      * @see hlf.module.Canvas#getAngDir
113      */
114     _generateSegments: function () {
115         var i, l, // counters
116             start, // start of ring
117             t, // start of current segment
118             dist, d, // total ring length and current length
119             seg, s, // base segment length and current length
120             ds, // segment flux modifier
121             br, // segment break length
122             dir = canvas.getAngDir(), // direction
123             _this = this;
124         /** @inner */
125         var setupBasePrc = function (type) {
126             // console.logArc(type == 'outer' ? _this : _this.innerRing);
127             // console.logArc(dist, 'distance for ' + type);
128             _this[type+'Segments'] = []; i = 0; d = 0; 
129             l = M.ceil(util.simpleRandom(2, C[type+'SegmentsMax']));
130             start = (type == 'outer') ? _this.angStart : _this.innerRing.angStart;
131             seg = dist / l;
132             br = M.sqrt(segmentFlux(type)) * C[type+'SegmentBreak'] * 
133                 M.max(M.pow(_this.lineWidth, 2), 1);
134         };
135         /** @inner */
136         var addSegmentPrc = function (type) { 
137             t = start + d * dir; 
138             d += s + br;
139             _this[type+'Segments'][i] = {
140                 'angStart': t, 
141                 'angEnd': t + s * dir
142             };
143             i += 1;
144         };
145         /** @inner */
146         var segmentFlux = function (type) {
147             return util.simpleRandom(C.segmentFluxMin, 
148                 C.segmentFluxMax);
149         };
150         /** @inner */
151         var updateSegment = function (type) {
152             ds = segmentFlux(type);
153             s = ds * seg - br;
154             if (type == 'outer') {
155                 s = M.max(s, M.H_PI);
156             }
157         };
158         /** @inner */
159         var canDrawSegment = function (type) {
160             updateSegment(type);
161             return dist > d + s + br;
162         };
163         if (this.hasOuterSegments()) {
164             dist = M.abs(this.angEnd - this.angStart);
165             setupBasePrc('outer');
166             while (i < l && canDrawSegment('outer')) {
167                 addSegmentPrc('outer');
168             }
169             // console.logAtPos('outerSegments', this); 
170         }
171         if (this.hasInnerSegments()) {
172             dist = M.abs(this.innerRing.angEnd - this.innerRing.angStart);
173             setupBasePrc('inner');
174             while (i < l && canDrawSegment('inner')) {
175                 addSegmentPrc('inner');
176             }
177             if (M.abs(dist - d) > M.PI / 8) {
178                 this.innerRing.hasEndBridge = false;
179             }
180             // console.logAtPos('innerSegments', this); 
181         }
182     },
183     // ----------------------------------------
184     // DRAW
185     // ----------------------------------------
186     /**
187      * Drawing controller for custom node. Because the paths drawn are
188      *      organized in a complex way, they must be drawn in a certain order,
189      *      almost like drawing with pen in hand.
190      * @see #_drawRing
191      * @see #_drawInnerRing
192      */
193     onDraw: function () {
194         this._drawRing();
195         if (this.hasInnerRing()) {
196             this._drawInnerRing();
197         }
198     },
199     // ----------------------------------------
200     // OUTER RING
201     // ----------------------------------------
202     /**
203      * Overrides the default implementation to allow segments. If ring has 
204      *      no segments, an alternative implementation is drawn.
205      * @see #_drawOuterRingSegment
206      * @see #_drawOuterRingBody
207      */
208     _drawRing: function () {
209         if (this.hasOuterSegments()) {
210             for (var i = 0, l = this.outerSegments.length;
211                 i < l; i += 1) {
212                 this._addGlow('_drawOuterRingSegment', null, 
213                     [this.outerSegments[i]]);
214             }
215         } else {
216             this._addGlow('_drawOuterRingBody');
217         }
218     },
219     /**
220      * Draws an arc as a new path and sets the stroke.
221      * @param {!Object number} segment
222      */
223     _drawOuterRingSegment: function (segment) {
224         context.beginPath();
225         canvas.arc(
226             this.pos.x, this.pos.y, this.rad,
227             segment.angStart, segment.angEnd, this.ang
228         );
229         context.stroke();        
230     },
231     /** Draws an arc as a new path and sets the stroke. */
232     _drawOuterRingBody: function () {
233         context.beginPath();
234         canvas.arc(
235             this.pos.x, this.pos.y, this.rad,
236             this.angStart, this.angEnd, this.ang
237         );
238         context.stroke();
239     },
240     // ----------------------------------------
241     // INNER RING
242     // ----------------------------------------
243     /**
244      * Main inner ring drawing procedure. If ring has no segments, 
245      *      an alternative implementation is drawn.
246      * @see #_drawInnerRingStart
247      * @see #_drawInnerRingSegment
248      * @see #_drawInnerRingBody
249      * @see #_drawInnerRingEnd
250      * @see innerGlow constant
251      */
252     _drawInnerRing: function () {
253         context.save();
254         context.lineWidth = this.innerRing.lineWidth;
255         if (this.innerRing.hasStartBridge) {
256             this._addGlow('_drawInnerRingStart', this.glowDist * C.innerGlow);
257         }
258         if (this.hasInnerSegments()) {
259             for (var i = 0, l = this.innerSegments.length;
260                 i < l; i += 1) {
261                 this._addGlow('_drawInnerRingSegment', this.glowDist * C.innerGlow, 
262                     [this.innerSegments[i]]);
263             }
264         } else {
265             this._addGlow('_drawInnerRingBody', this.glowDist * C.innerGlow);
266         }
267         if (this.innerRing.hasEndBridge) {
268             this._addGlow('_drawInnerRingEnd', this.glowDist * C.innerGlow);
269         }
270         context.restore();
271     },
272     /** Draws start bridge as a new path and sets the stroke. */
273     _drawInnerRingStart: function () {
274         var r = this.innerRing;
275         context.beginPath();
276         context.lineTo(
277             // outer ring start point
278             this.pos.x + this.rad * M.cos(r.angStart + this.ang + r.bridgeBend), 
279             this.pos.y + this.rad * M.sin(r.angStart + this.ang + r.bridgeBend) 
280         );
281         context.quadraticCurveTo(
282             // inner ring start point
283             this.pos.x + r.rad * M.cos(r.angStart + this.ang + r.bridgeBend),
284             this.pos.y + r.rad * M.sin(r.angStart + this.ang + r.bridgeBend),
285             this.pos.x + r.rad * M.cos(r.angStart + this.ang),
286             this.pos.y + r.rad * M.sin(r.angStart + this.ang)
287         );
288         context.stroke();
289     },
290     /** Draws an arc as a new path and sets the stroke. */
291     _drawInnerRingBody: function () {
292         context.beginPath();
293         canvas.arc(
294             this.pos.x, this.pos.y, this.innerRing.rad,
295             this.innerRing.angStart, this.innerRing.angEnd, this.ang
296         );
297         context.stroke();
298     },
299     /**
300      * Draws an arc as a new path and sets the stroke.
301      * @param {!Object number} segment
302      */
303     _drawInnerRingSegment: function (segment) {
304         context.beginPath();
305         canvas.arc(
306             this.pos.x, this.pos.y, this.innerRing.rad,
307             segment.angStart, segment.angEnd, this.ang
308         );
309         context.stroke();
310     },
311     /** Draws end bridge as a new path and sets the stroke. */
312     _drawInnerRingEnd: function () {
313         var r = this.innerRing;
314         context.beginPath();
315         context.lineTo(
316             this.pos.x + r.rad * M.cos(r.angEnd + this.ang),
317             this.pos.y + r.rad * M.sin(r.angEnd + this.ang)
318         );
319         context.quadraticCurveTo(
320             // outer ring end point
321             this.pos.x + r.rad * M.cos(r.angEnd + this.ang - r.bridgeBend),
322             this.pos.y + r.rad * M.sin(r.angEnd + this.ang - r.bridgeBend),
323             this.pos.x + this.rad * M.cos(r.angEnd + this.ang - r.bridgeBend),
324             this.pos.y + this.rad * M.sin(r.angEnd + this.ang - r.bridgeBend)
325         );
326         context.stroke();
327     },
328     // ----------------------------------------
329     // ANIMATE
330     // ----------------------------------------
331     /** @ignore */
332     toString: function () {
333         return pkg + 'sakuraDrops.DropNode';
334     }    
335 });
336 }); // namespace
