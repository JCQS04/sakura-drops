  8 Namespace(pkg + 'sakuraDrops');
  9 Namespace.use(pkg + '*', function () {
 10 var app = sakuraDrops, 
 11     C = app.constants,
 12     M = app.Math, 
 13     canvas, context,
 14     /** 
 15      * A hack to update the scope globals for canvas when canvas is ready. 
 16      *      This needs to be called in the constructor of every class and 
 17      *      included in files needing the app's scope globals.
 18      */
 19     requireCanvasGlobals = function () {
 20         if (!canvas) {
 21             canvas = app.canvas;
 22             context = canvas.context;
 23         }  
 24     };
 25 /**
 26  * @class Base functionality shared by all nodes in this app.
 27  * @augments hlf.util.BaseClass
 28  * @augments hlf.module.EventMixin
 29  * @property {number} ang Rotation offset.
 30  * @property {number} luck Chance of certain traits.
 31  * @property {number} lineWidth Base stroke size, if applicable.
 32  * @property {number} glowDist Size of glowing stroke, related to base stroke.
 33  * @property {number} angStart Where arc starts, in radians.
 34  * @property {number} angEnd Where arc ends, in radians.
 35  * @property {boolean} inBounds Requirement to be entirely visible on canvas.
 36  * @property {boolean} isAwake Waking behavior, generally waking on user
 37  *      interaction and returning to sleep after a period of inactivity.
 38  * @property {number} spinAnimation Animation id in the canvas' index.
 39  * @param {!Object} params Parameters as key-value pairs:
 40  *      <br/>pos Defaults to current canvas pen position.
 41  *      <br/>ang Defaults to 0.
 42  *      <br/>luck Defaults to random.
 43  *      <br/>rad Defaults to {@link .BASE_RAD}.
 44  *      <br/>lineWidth Defaults to {@link .BASE_LINEWIDTH}.
 45  */
 46 app.BaseNode = util.Circle.extend(util.extend({
 47     /** @lends app.BaseNode# */
 48     ang: undefined,
 49     luck: undefined,
 50     lineWidth: undefined,
 51     glowDist: undefined,
 52     angStart: undefined,
 53     angEnd: undefined,
 54     inBounds: undefined,
 55     isAwake: undefined,
 56     spinAnimation: undefined,
 57     // ----------------------------------------
 58     // ACCESSORS
 59     // ----------------------------------------
 60     /**
 61      * Uses the ease-in-out-cubic easing equation to get the d(change in value).
 62      * @param {!int} t Current time, in millis.
 63      * @param {!number} b Beginning value.
 64      * @param {!number} c Overall change in value.
 65      * @param {!int} d Duration, in millis.
 66      * @return {number} Size of current step.
 67      */
 68     getSpinStep: function (t, b, c, d) {
 69         if ((t /= d / 2) < 1) {
 70             return c / 2 * t * t * t + b;
 71         }
 72 		return c / 2 * ((t -= 2) * t * t + 2) + b;
 73     },
 74     /**
 75      * Accessor.
 76      * @return {number}
 77      */
 78     getArcLen: function () {
 79         return M.abs(this.angEnd - this.angStart);
 80     },
 81     // ----------------------------------------
 82     // METHODS
 83     // ----------------------------------------
 84     /**
 85      * Sets up node and does some initial actions. Supports delegation.
 86      * @see #update
 87      * @see #didCreate
 88      */
 89     _construct: function (params) {
 90         requireCanvasGlobals();
 91         this.pos = {};
 92         this.pos.x = util.toInt(params.pos.x) || canvas.getX();
 93         this.pos.y = util.toInt(params.pos.y) || canvas.getY();
 94         this.ang = 0;
 95         this.rad = params.rad || C.BASE_NODE.rad;
 96         this.luck = util.isNumber(params.luck) ? params.luck : M.random();
 97         this.lineWidth = params.lineWidth || C.BASE_NODE.lineWidth;
 98         this.glowDist = M.pow(this.lineWidth, 4) + C.BASE_NODE.glowDistance;
 99         this.angStart = params.angStart || 0;
100         this.angEnd = params.angEnd || PI * 2;
101         this.inBounds = params.inBounds || false;
102         this.isAwake = false;
103         this.update();
104         this.didCreate();
105     },
106     // ----------------------------------------
107     // DELEGATES
108     // ----------------------------------------
109     /**#@+
110        Delegate method container; extend and fill as needed.
111     */
112     didCreate: function () {},
113     didUpdate: function () {},
114     didWake: function () {},
115     didSleep: function () {},
116     onDraw: function () {},
117     onDrawRing: function () {},
118     onSpin: function () {},
119     /**#@-*/
120     // ----------------------------------------
121     // UPDATE BASE NODE
122     // ----------------------------------------
123     /**
124      * Updating API. Updates the node in the context of its surroundings.
125      * @see #_stayInBounds
126      * @see #didUpdate
127      */ 
128     update: function () {
129         if (this.inBounds) {
130             this._stayInBounds();
131         }
132         this.didUpdate();
133     },
134     /** Stay within the canvas, adjusted for radius and stroke. */
135     _stayInBounds: function () {
136         var d = this.rad + this.lineWidth * 3;
137         this.pos.x = util.toInt(util.constrain(this.pos.x, d, canvas.getWidth() - d));
138         this.pos.y = util.toInt(util.constrain(this.pos.y, d, canvas.getHeight() - d));
139     },
140     // ----------------------------------------
141     // DRAW BASE NODE
142     // ----------------------------------------
143     /**
144      * Drawing API. Moves the pen first. Supports delegation. 
145      * @see #onDraw
146      */
147     draw: function () {
148         canvas.movePlotter(this.pos.x, this.pos.y);
149         this.onDraw();
150     },
151     /**
152      * Root ring drawing function. Sets the stroke and adds glow in own context.
153      * @see #_addGlow
154      */
155     _drawRing: function () {
156         context.save();
157         context.lineWidth = this.lineWidth;
158         this._addGlow('onDrawRing');
159         context.restore();
160     },
161     /**
162      * Adds glow to any path, which gets drawn multiple times in own context. 
163      * @param {!function(array)} drawLayerCallback Required drawing socket.
164      * @param {?number=} glowDist Optional custom distance.
165      * @param {?Array=} callbackArgs Parameters for the socket as needed.
166      */
167     _addGlow: function (drawLayerSocket, glowDist, callbackArgs) {
168         context.save();
169         glowDist = glowDist || this.glowDist;
170         var d = 0;
171         for (var i = 0; i < glowDist; i += 1) {
172             d = 1 + M.pow(i/glowDist, 2);
173             context.lineWidth += d;
174             this[drawLayerSocket].apply(this, callbackArgs || []);
175         }
176         context.restore();
177     },
178     // ----------------------------------------
179     // BEHAVIORS
180     // ----------------------------------------
181     /**
182      * Naturally activate other behaviors and node.
183      */
184     wake: function () {
185         if (this.isAwake) {
186             // return;
187         }
188         this.isAwake = true;
189         this.startSpin();
190         this.didWake();
191         // console.logAtPos('woke', this); 
192     },
193     /**
194      * Naturally deactivate other behaviors and node.
195      */
196     sleep: function () {
197         if (!this.isAwake) {
198             // return;
199         }
200         this.isAwake = false;
201         this.stopSpin();
202         this.didSleep();
203         // console.logAtPos('slept', this); 
204     },
205     /**
206      * @name app.BaseNode#didAnimationStep
207      * @event
208      * @param {Event} e
209      */
210     /**
211      * Starts spin animation and set binding as needed. Currently spins
212      *      for a pseudorandom distance with easing.
213      * @see #getSpinStep
214      */
215     startSpin: function () {
216         var _this = this,
217             beginning = this.ang,
218             change = M.PI * util.simpleRandom(C.BASE_NODE.spinMin, C.BASE_NODE.spinMax),
219             duration = C.BASE_NODE.spinSpeed,
220             callback = function (elapsed, complete) {
221                 if (complete) {
222                     _this.sleep();
223                 } else {
224                     _this.update();
225                     _this.ang = _this.getSpinStep(elapsed, beginning, change, duration);
226                     _this.onSpin();
227                     _this.trigger('didAnimationStep');
228                     // console.log('animationStep');
229                 }
230             };
231         if (!this.spinAnimation) {
232             this.spinAnimation = canvas.animate(null, callback, duration);
233         } else {
234             canvas.animate(null, callback, duration, this.spinAnimation);
235         }
236     },
237     /**
238      * Stops spinning animation.
239      */
240     stopSpin: function () {
241         canvas.pauseAnimation(this.spinAnimation);
242     },
243     /** @ignore */
244     toString: function () {
245         return pkg + 'sakuraDrops.BaseNode';
246     }
247 }, module.EventMixin));

251 /**
252  * @class Base functionality shared by all node managers in this app.
253  * @augments hlf.util.BaseClass
254  * @augments hlf.module.EventMixin
255  * @augments hlf.util.CanvasEventMixin
256  * @property {jQuery} $canvas Canvas jQuery element, required for event delegation.
257  * @property {Array hlf.sakuraDrops.BaseNode} nodes Index of all nodes.
258  * @property {Object} params Bin for storing all the manager settings.
259  * @property {boolean} unitTest For testing, only run one node.
260  * @property {Object boolean} ready State flags useful for event handlers.
261  * @param {!Object} params Settings as key-value pairs:
262  *      <br/>num Required.
263  *      <br/>unitTest Defaults to false.
264  */
265 app.BaseManager = util.BaseClass(util.extend(util.CanvasEventMixin, {
266     /** @lends app.BaseManager# */
267     $canvas: undefined,
268     nodes: undefined,
269     params: undefined,
270     unitTest: undefined,
271     ready: undefined,
272     /**
273      * @see #didCreate
274      * @see #_populate
275      */
276     _construct: function (params) {
277         requireCanvasGlobals();
278         this.params = params;
279         this.$canvas = $(canvas.canvas);
280         this.unitTest = params.unitTest || false;
281         this._populate();
282         this.bindMouse();
283         this.ready = {};
284         this.didCreate();
285     },
286     // ----------------------------------------
287     // DELEGATES
288     // ----------------------------------------
289     /**#@+
290        Delegate method container; extend and fill as needed. 
291        Return true for will and did delegates to break from the procedure.
292     */
293     onPopulate: function () {},
294     onUpdate: function () {},
295     didCreate: function () {},
296     didDraw: function () {},
297     didLoad: function () {},
298     /**#@-*/
299     // ----------------------------------------
300     // SETUP NODES
301     // ----------------------------------------
302     /**
303      * Runs a loop to instantiate the nodes, including binding to
304      *      node events.
305      * @see hlf.sakuraDrops.BaseNode#didAnimationStep
306      * @see #onPopulate
307      */ 
308     _populate: function () {
309         var _this = this;
310         if (this.unitTest) {
311             this.params.num = 1;
312         }
313         this.nodes = [];
314         for (var i = 0; i < this.params.num; i += 1) {
315             this.nodes[i] = this.onPopulate(i);
316             this.nodes[i].bind('didAnimationStep', function () {
317                 _this.draw();
318             });
319         }        
320     },
321     /**
322      * Updating API. 
323      * @see #onUpdate
324      */
325     update: function () {
326         this.onUpdate();
327     },
328     // ----------------------------------------
329     // DRAW NODES
330     // ----------------------------------------
331     /**
332      * Sets the canvas context to use the default theme, which is 
333      *      translucent white fill and stroke with round caps.
334      */ 
335     theme: function () {
336         context.lineCap = 'round';
337         context.fillStyle = 'rgba(255,255,255, .2)';
338         context.strokeStyle = 'rgba(255,255,255, .2)';            
339     },
340     /**
341      * Drawing API. Runs a loop to use each node's drawing API.
342      * @see #didDraw
343      * @see #theme
344      */
345     draw: function () {
346         this.theme();
347         for (var i = 0, l = this.nodes.length; i < l; i += 1) {
348             this.nodes[i].draw();
349         }
350         this.didDraw();
351     },
352     // ----------------------------------------
353     // INPUT
354     // ----------------------------------------
355     /**
356      * Wakes the node if needed when mouse moves inside the canvas. The 
357      *      index position of the focused node is updated. The previous node
358      *      is put to sleep, if needed. Runs a loop and exits upon finding 
359      *      active node.
360      * @param {!Event} evt jQuery mousemove event.
361      * @todo Disabled sleep on focus out until slowSpin exists.
362      */
363     onMouseMove: function (evt) {
364         if (!this.ready.mouseMove) {
365             return;
366         }
367         // console.log('onMouseMove');
368         var _this = this;
369         for (var i = 0, l = this.nodes.length; i < l; i += 1) {
370             if (this._contains(this.nodes[i], {x: evt.offsetX, y: evt.offsetY})) {
371                 // moved from blank space
372                 this.nodes[i].wake();
373                 this.ready.mouseMove = false;
374                 setTimeout(function () {
375                     _this.ready.mouseMove = true;
376                 }, C.MOUSEMOVE_TIMEOUT)
377                 break;
378             }
379         }
380     },    
381     // ----------------------------------------
382     // UTILITY
383     // ----------------------------------------
384     /**
385      * Utility function to check if a node is on a position.
386      * @param {!hlf.sakuraDrops.BaseNode} node 
387      * @param {!Object number} pos 
388      * @return {boolean}
389      */
390     _contains: function(node, pos) {
391         var dx = node.pos.x - pos.x,
392             dy = node.pos.y - pos.y;
393         return util.dist(dx, dy) <= node.rad;
394     },
395     /**
396      * Utility function to check if nodes overlap.
397      * @param {!hlf.sakuraDrops.BaseNode} node1 
398      * @param {!hlf.sakuraDrops.BaseNode} node2 
399      * @return {boolean}
400      */
401     _intersects: function(node1, node2) {
402         var dx = node2.pos.x - node1.pos.x,
403             dy = node2.pos.y - node1.pos.y,
404             d = util.dist(dx, dy);
405         return d < node1.rad || d < node2.rad;
406     },
407     /** @ignore */
408     toString: function () {
409         return pkg + 'sakuraDrops.BaseManager';
410     }
411 }, module.EventMixin));
412 }); // namespace
