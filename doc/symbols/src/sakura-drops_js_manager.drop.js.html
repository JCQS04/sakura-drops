1 /**
  2  * @fileoverview Drop nodes manager class for Sakura Drops app.
  3  * @author peng@pengxwang.com (Peng Wang)
  4  * @requires Namespace
  5  * @requires hlf.util
  6  * @requires hlf.module.Canvas
  7  * @requires hlf.sakuraDrops.DropNode
  8  * @requires hlf.sakuraDrops.BaseManager
  9  */
 10 // ----------------------------------------
 11 // OVERHEAD
 12 // ----------------------------------------
 13 Namespace(pkg + 'sakuraDrops');
 14 Namespace.use(pkg + '*', function () {
 15 var app, C, M, canvas, context;
 16 var setScopeGlobals = function () {
 17     if (canvas) {
 18         return;
 19     }
 20     app = sakuraDrops;
 21     C = app.constants;
 22     M = app.Math;
 23     canvas = app.canvas;
 24     context = app.context;
 25 }; 
 26 setScopeGlobals();
 27 // ----------------------------------------
 28 // CLASS
 29 // ----------------------------------------
 30 /**
 31  * @class Manager for DropNodes. Responsible for setup and interaction
 32  *      specific to node instances.
 33  * @augments hlf.sakuraDrops.BaseManager
 34  * @property {hlf.module.CirclePacker} cp
 35  */
 36 app.DropManager = app.BaseManager.extend({
 37     /** @lends app.DropManager# */
 38     /** @type {module.CirclePacker} */
 39     cp: undefined,
 40     // ----------------------------------------
 41     // ACCESSORS
 42     // ----------------------------------------
 43     /**
 44      * Accessor for center of circle packer. Current center of canvas.
 45      * @return {Object number}
 46      */
 47     getAttractorPos: function () {
 48         return {
 49             x: canvas.getWidth() / 2,
 50             y: canvas.getHeight() / 2
 51         };
 52     },    
 53     // ----------------------------------------
 54     // SETUP
 55     // ----------------------------------------
 56     /**
 57      * Sets up node at a random position with pseudo-random size that affects
 58      *      line width, and pseudorandom length. Note luck is not customized.
 59      * @param {!int} i Index for node.
 60      * @see hlf.util.simpleRandom
 61      * @see hlf.util.bufferedRandom
 62      * @see hlf.util.curvingBufferedRandom
 63      */
 64     onPopulate: function (i) {
 65         setScopeGlobals();
 66         var x = util.simpleRandom(canvas.getWidth()),
 67             y = util.simpleRandom(canvas.getHeight()),
 68             rad = util.curvingBufferedRandom(C.DROP_NODE.rad, .5, 2),
 69             lineWidth = C.DROP_NODE.lineWidth * rad / C.DROP_NODE.rad,
 70             angStart = util.simpleRandom(M.TWO_PI) + M.TWO_PI,
 71             angEnd = angStart + util.bufferedRandom(M.TWO_PI, 2) * 
 72                 canvas.getAngDir(),
 73             params = { 'pos': {'x': x, 'y': y}, 'rad': rad, 'lineWidth': lineWidth,
 74                 'angStart': angStart, 'angEnd': angEnd };
 75         if (this.unitTest) {
 76             params.luck = 0; // activate all side cases
 77         }
 78         return app.DropNode.create(params);
 79     },
 80     /**
 81      * Sets up the circle packer and binds its drawing socket to the drawing API.
 82      * @see hlf.module.CirclePacker#drawingSocket
 83      * @see #draw
 84      */
 85     didCreate: function () {
 86         var _this = this;
 87         this.cp = module.CirclePacker.create(this.nodes, 
 88             this.getAttractorPos(), C.CIRCLE_PACKER.passes);
 89         this.cp.bind('drawingSocket', function () {
 90             canvas.background('rgb(0,0,0)');
 91             _this.draw();
 92         });
 93     },
 94     /** Runs / refreshes the circle packer. */
 95     onUpdate: function () {
 96         this.cp.run();
 97     },
 98     /** @ignore */
 99     toString: function () {
100         return pkg + 'sakuraDrops.DropManager';
101     }
102 });
103 }); // namespace
