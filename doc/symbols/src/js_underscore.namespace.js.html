1 /**
  2  * @fileoverview Light port of NamespaceJS for UnderscoreJS.
  3  * @author Maxime Bouroumeau-Fuseau (original)
  4  * @author Peng Wang 
  5  */
  6 (function () {
  7   /**
  8    * Creates an Object following the specified namespace identifier.
  9    * @param {string} identifier The namespace string.
 10    * @param {?Object=} classes An object which properties will be added to the namespace.
 11    * @return {Object} The most inner object
 12    */
 13   _.namespace = function (identifier) {
 14     var classes = arguments[1] || false;
 15     var ns = window;
 16     if (identifier !== '') {
 17       var parts = identifier.split(_.namespacer);
 18       for (var i = 0; i < parts.length; i++) {
 19         if (!ns[parts[i]]) {
 20           ns[parts[i]] = {};
 21         }
 22         ns = ns[parts[i]];
 23       }
 24     }
 25     if (classes) {
 26       for (var c in classes) {
 27         if (classes.hasOwnProperty(c)) {
 28           ns[c] = classes[c];
 29         }
 30       }
 31     }
 32     return ns;
 33   };
 34   /**
 35    * Imports properties from the specified namespace to the global space (window).
 36    *      The identifier string can contain the * wildcard character as its last 
 37    *      segment (eg: com.test.*) which will import all properties from the namespace.
 38    *      If not, the targeted namespace will be imported (ie. if com.test is 
 39    *      imported, the test object will now be global). 
 40    * @public
 41    * @param {string} identifier The namespace string.
 42    * @param {?function=} callback A function to call when the process is completed
 43    */
 44   _.using = function (identifier) {
 45     var identifiers = identifier.pop ? identifier : [identifier];
 46     var callback = arguments[1] || false;
 47     var event = { 'identifier': identifier };
 48     var parts, target, ns;
 49     for (var i = 0; i < identifiers.length; i++) {
 50       identifier = identifiers[i];
 51       parts = identifier.split(_.namespacer);
 52       target = parts.pop();
 53       ns = _.namespace(parts.join(_.namespacer));
 54       if (target == '*') {
 55         // imports all objects from the identifier, can't use include() in that case
 56         for (var objectName in ns) {
 57           if (ns.hasOwnProperty(objectName)) {
 58             window[objectName] = ns[objectName];
 59           }
 60         }
 61       } else if (ns[target]) {
 62         // imports only one object
 63         window[target] = ns[target];
 64       }
 65     }
 66     // all identifiers have been unpacked
 67     if (typeof callback === "function") { 
 68       callback(); 
 69     }  
 70   };
 71   /**
 72    * Namespace segment separator
 73    * @var String
 74    */
 75   _.namespacer = '.';    
 76 })();
